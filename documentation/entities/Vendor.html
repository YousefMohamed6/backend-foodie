import {
  Column,
  CreateDateColumn,
  Entity,
  Index,
  JoinColumn,
  OneToOne,
  PrimaryGeneratedColumn,
  UpdateDateColumn,
} from 'typeorm';
import { User } from '../../users/entities/user.entity';

@Entity('vendors')
export class Vendor {
  @PrimaryGeneratedColumn('uuid')
  id: string;

  @Column()
  @Index()
  authorId: string;

  @OneToOne(() => User)
  @JoinColumn({ name: 'authorId' })
  author: User;

  @Column()
  title: string;

  @Column('text', { nullable: true })
  description: string;

  @Column({ nullable: true })
  photo: string;

  @Column({ nullable: true })
  logo: string;

  @Column('json', { name: 'categoryID', nullable: true })
  categoryIds: string[];

  @Column('json', { nullable: true })
  categoryTitle: string[];

  @Column({ nullable: true })
  vendorTypeId: string;

  @Column('decimal', { precision: 10, scale: 8 })
  @Index()
  latitude: number;

  @Column('decimal', { precision: 11, scale: 8 })
  @Index()
  longitude: number;

  @Column({ nullable: true })
  address: string;

  // Note: Spatial types require PostGIS extension.
  // For simplicity in this native implementation without complex setup,
  // we'll stick to lat/long columns for now, but keeping the field structure ready.
  // @Column({
  //   type: 'point',
  //   spatialFeatureType: 'Point',
  //   nullable: true,
  // })
  // @Index({ spatial: true })
  // coordinates: any;

  @Column({ nullable: true })
  @Index()
  zoneId: string;

  @Column({ name: 'phonenumber', nullable: true })
  phoneNumber: string;

  @Column({ nullable: true })
  fcmToken: string;

  @Column({ name: 'reststatus', default: true })
  restStatus: boolean;

  @Column({ name: 'dine_in_active', default: false })
  dineInActive: boolean;

  @Column({ nullable: true })
  openDineTime: string;

  @Column({ nullable: true })
  closeDineTime: string;

  @Column({ default: false })
  enabledDiveInFuture: boolean;

  @Column('decimal', { precision: 10, scale: 2, default: 0 })
  walletAmount: number;

  @Column('decimal', { precision: 10, scale: 2, default: 0 })
  reviewsSum: number;

  @Column({ default: 0 })
  reviewsCount: number;

  @Column('json', { name: 'adminCommission', nullable: true })
  adminCommission: any;

  @Column('json', { nullable: true })
  workingHours: any[];

  @Column('json', { nullable: true })
  specialDiscount: any[];

  @Column({ default: false })
  specialDiscountEnable: boolean;

  @Column('json', { name: 'DeliveryCharge', nullable: true })
  deliveryCharge: any;

  @Column('json', { nullable: true })
  filters: any;

  @Column('json', { nullable: true })
  restaurantMenuPhotos: string[];

  @Column('json', { nullable: true })
  photos: string[];

  @Column({ nullable: true })
  restaurantCost: string;

  @Column({ default: false })
  hidephotos: boolean;

  @Column({ nullable: true })
  subscriptionPlanId: string;

  @Column({ nullable: true })
  subscriptionExpiryDate: Date;

  @Column({ nullable: true })
  subscriptionTotalOrders: string;

  @Column({ default: false })
  isSelfDelivery: boolean;

  @CreateDateColumn()
  createdAt: Date;

  @UpdateDateColumn()
  updatedAt: Date;
}
